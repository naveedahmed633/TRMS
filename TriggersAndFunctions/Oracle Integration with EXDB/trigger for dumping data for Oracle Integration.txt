CreateTRIGGER [dbo].[PopulateHaTransit]
   ON [EXDB].[dbo].[exdb]     -- Trigger fires after insert on HaTransits table
   AFTER INSERT
AS 
BEGIN
    -- Insert the newly inserted data from HaTransits into HaTransits_Dump
    INSERT INTO HaTransits_Dump(
        C_Name, C_Unique, C_Date, C_Time, L_UID, L_TID, active, mode, Device_Name
    )
    -- Select data from the inserted pseudo-table and join with Terminals to get Device_Name
    SELECT
        HT.user_name,          -- user_name from HaTransits
        HT.unique_id,          -- unique_id from HaTransits
        HT.date,               -- Date from HaTransits
        -- Convert HT.time (VARCHAR(6)) to TIME, and format it to remove fractional seconds
        FORMAT(CONVERT(TIME, STUFF(STUFF(HT.time, 3, 0, ':'), 6, 0, ':'), 108), 'HH:mm:ss') AS C_Time,  
        HT.user_id,            -- user_id from HaTransits
        HT.terminal_id,        -- terminal_id from HaTransits
        1,                     -- active status from HaTransits
        1,                     -- mode from HaTransits
        T.name                 -- device name from UCDB.dbo.Terminals
    FROM inserted HT
    -- Join with UCDB.dbo.Terminals to get the Device_Name where L_TID matches
    LEFT JOIN UCDB.dbo.Terminals T
    ON HT.terminal_id = T.Terminal_ID;  -- Assuming Terminal_ID is the column in Terminals for the device id
END
GO
