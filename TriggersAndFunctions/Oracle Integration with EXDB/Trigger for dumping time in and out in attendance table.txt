Create TRIGGER [dbo].[PopulateAttendance]
ON [dbo].[HaTransits_Dump]
AFTER INSERT
AS
BEGIN
    SET NOCOUNT ON;

    -- Declare variables to store incoming data
    DECLARE @User_ID INT, @Date DATE, @Time_in TIME, @Time_out TIME, @Device_IN_Name VARCHAR(100), @Device_OUT_Name VARCHAR(100);

    -- Cursor to loop through inserted records
    DECLARE attendance_cursor CURSOR FOR
    SELECT 
        in_data.C_Unique AS User_ID,
        in_data.C_Date AS Date,
        MIN(in_data.C_Time) AS Time_in,
        MAX(out_data.C_Time) AS Time_out,
        in_data.Device_Name AS Device_IN_Name,
        out_data.Device_Name AS Device_OUT_Name
    FROM HaTransits_Dump in_data
    LEFT JOIN HaTransits_Dump out_data
        ON in_data.C_Unique= out_data.C_Unique
       AND in_data.C_Date = out_data.C_Date
       AND in_data.Device_Name = 'Entry'
       AND out_data.Device_Name = 'Exit'
    WHERE in_data.Device_Name = 'Entry'
    GROUP BY 
        in_data.C_Unique, 
        in_data.C_Date, 
        in_data.Device_Name, 
        out_data.Device_Name;
    OPEN attendance_cursor;

    FETCH NEXT FROM attendance_cursor INTO @User_ID, @Date, @Time_in, @Time_out, @Device_IN_Name, @Device_OUT_Name;

    WHILE @@FETCH_STATUS = 0
    BEGIN
        -- Check if record exists in Attendance table
        IF EXISTS (SELECT 1 FROM dbo.Consolidate_Attendance WHERE User_ID = @User_ID AND Date = @Date)
        BEGIN
            -- Update existing record
            UPDATE dbo.Consolidate_Attendance
            SET 
                Time_in = @Time_in,
                Time_out = @Time_out,
                Device_IN_Name = @Device_IN_Name,
                Device_OUT_Name = @Device_OUT_Name
            WHERE User_ID = @User_ID AND Date = @Date;
        END
        ELSE
        BEGIN
            -- Insert new record if it does not exist
            INSERT INTO dbo.Consolidate_Attendance (User_ID, Date, Time_in, Time_out, Device_IN_Name, Device_OUT_Name)
            VALUES (@User_ID, @Date, @Time_in, @Time_out, @Device_IN_Name, @Device_OUT_Name);
        END;

        FETCH NEXT FROM attendance_cursor INTO @User_ID, @Date, @Time_in, @Time_out, @Device_IN_Name, @Device_OUT_Name;
    END;

    CLOSE attendance_cursor;
    DEALLOCATE attendance_cursor;
END;
