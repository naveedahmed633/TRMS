-- Define the number of rows you want to generate
DECLARE @RowCount INT = 2000; -- Adjust this for the number of rows you need

-- Initialize the counter
DECLARE @Counter INT = 0;

-- Start the loop
WHILE @Counter < @RowCount
BEGIN
    -- Declare variables for random employee details
    DECLARE @RandomUserId INT;
    DECLARE @RandomUserName NVARCHAR(50);
    DECLARE @RandomEmployeeId NVARCHAR(50);
    DECLARE @RandomTime NVARCHAR(10); -- Declare variable for C_time

    -- Fetch a random employee from the employees table
    SELECT TOP 1 
        @RandomUserId = EmployeeID, 
        @RandomUserName = first_name, 
        @RandomEmployeeId = employee_code
    FROM employees where active = 1
    ORDER BY NEWID();

    -- Generate a random time between 9:00 AM and 5:00 PM
    SELECT @RandomTime = CONVERT(NVARCHAR(10), 
                                 DATEADD(SECOND, ABS(CHECKSUM(NEWID())) % 28800, '09:00:00'), 
                                 108);

    -- Check if a random employee was successfully fetched
    IF @RandomUserId IS NOT NULL
    BEGIN
        -- Insert the random data into the attendance table
        INSERT INTO trms.dbo.HaTransits(
            C_name, C_unique, C_date, C_time, L_UId, active, course_active, L_TID, mode
        )
        VALUES (
            -- C_name: Randomly picked username from the employees table
            @RandomUserName,
            -- C_unique: Employee ID from the employees table
            @RandomEmployeeId,
            -- C_date: Today's date
            CAST(GETDATE() AS DATE),
            -- C_time: Random time as NVARCHAR
            @RandomTime,
            -- L_UId: Employee ID (now as INT)
            @RandomUserId,
            -- active: Fixed value 1
            1,
            -- course_active: Fixed value 1
            1,
            -- L_TID: Fixed value 1
            1,
            -- mode: Fixed value 0
            0
        );

        -- Increment the counter
        SET @Counter = @Counter + 1;
    END
    ELSE
    BEGIN
        -- Exit the loop if no employees are found
        PRINT 'No employees found in the employees table. Exiting...';
        BREAK;
    END
END;
