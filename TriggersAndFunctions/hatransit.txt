/************************** Rectify Ha transit entries **********************************/
update 
	A
set
	A.C_Time = SUBSTRING(B.C_Time,0,9)
from
	TimeTune.dbo.HaTransits as A
	INNER JOIN TimeTune.dbo.HaTransits as B
		ON A.HaTransitId = B.HaTransitId
