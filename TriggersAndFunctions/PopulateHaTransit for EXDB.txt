ALTER TRIGGER [dbo].[PopulateHaTransit]
   ON  [EXDB].[dbo].[exdb]
   AFTER insert
AS 
BEGIN

	declare @status int =1

	insert into TRMS.[dbo].[HaTransits] (C_Name,C_Unique,C_Date,C_Time,L_UID,L_TID,active,mode)
		
select  [user_name],[unique_id],
CAST(CAST(event_time AS DATE) AS DATETIME) AS date_part,
    FORMAT(CAST(event_time AS DATETIME), 'HH:mm:ss') AS time_part,
[user_id],[terminal_id],@status,func
from inserted
		where user_id >-1 

END
